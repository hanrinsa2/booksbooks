CREATE TABLE Member
(Member_nickname CHAR(15) COMMENT '회원 닉네임',
Member_password CHAR(20) COMMENT '회원 비밀번호',
Member_name CHAR(10) COMMENT '회원 이름',
PRIMARY KEY (Member_nickname)
);

CREATE TABLE MainStory
(Story_no INT NOT NULL AUTO_INCREMENT COMMENT '스토리 번호',
Story_name CHAR(20) COMMENT '스토리 이름',
Story_content MEDIUMTEXT COMMENT '스토리 내용',
Story_writer CHAR(10) COMMENT '스토리 작가',
Story_grade_avg FLOAT(2,1) DEFAULT 0 COMMENT '스토리 평점',	
PRIMARY KEY (Story_no)
);


CREATE TABLE Grade_MainStory
(Member_nickname_GS CHAR(15) COMMENT '회원 닉네임',
Story_no_GS INT NOT NULL COMMENT '스토리 번호',
Story_grade FLOAT COMMENT '스토리 점수',
PRIMARY KEY (Member_nickname_GS, Story_no_GS),
FOREIGN KEY (Member_nickname_GS) REFERENCES Member(Member_nickname) ON DELETE CASCADE ON UPDATE CASCADE,
FOREIGN KEY (Story_no_GS) REFERENCES MainStory(Story_no) ON DELETE CASCADE ON UPDATE CASCADE
);

CREATE TABLE MainPoem
(Poem_no INT NOT NULL AUTO_INCREMENT COMMENT '시 번호',
Poem_name CHAR(20) COMMENT '시 이름',
Poem_content MEDIUMTEXT COMMENT '시 내용',
Poem_writer CHAR(10) COMMENT '시 작가',
Poem_grade_avg FLOAT(2,1) DEFAULT 0 COMMENT '시 평점',	
PRIMARY KEY (Poem_no)
);


CREATE TABLE Grade_MainPoem
(Member_nickname_GP CHAR(15) COMMENT '회원 닉네임', 
Poem_no_GP INT NOT NULL COMMENT '시 번호',
Poem_grade FLOAT COMMENT '시 점수',
PRIMARY KEY (Member_nickname_GP, Poem_no_GP),
FOREIGN KEY (Member_nickname_GP) REFERENCES Member(Member_nickname) ON DELETE CASCADE ON UPDATE CASCADE,
FOREIGN KEY (Poem_no_GP) REFERENCES MainStory(Poem_no) ON DELETE CASCADE ON UPDATE CASCADE
);

CREATE TABLE MainEssay
(Essay_no INT NOT NULL AUTO_INCREMENT COMMENT '수필 번호',
Essay_name CHAR(20) COMMENT '수필 이름',
Essay_content MEDIUMTEXT COMMENT '수필 내용',
Essay_writer CHAR(10) COMMENT '수필 작가',
Essay_grade_avg FLOAT(2,1) DEFAULT 0 COMMENT '수필 평점',	
PRIMARY KEY (Essay_no)
);


CREATE TABLE Grade_MainEssay
(Member_nickname_GE CHAR(15) COMMENT '회원 닉네임', 
Essay_no_GE INT NOT NULL COMMENT '수필 번호',
Essay_grade FLOAT COMMENT '수필 점수',
PRIMARY KEY (Member_nickname_GE, Essay_no_GE),
FOREIGN KEY (Member_nickname_GE) REFERENCES Member(Member_nickname) ON DELETE CASCADE ON UPDATE CASCADE,
FOREIGN KEY (Essay_no_GE) REFERENCES MainStory(Essay_no) ON DELETE CASCADE ON UPDATE CASCADE
);


CREATE TABLE Category
(Category_no INT NOT NULL COMMENT '카테고리 번호',
Category_type CHAR(10) COMMENT '카테고리 종류',
PRIMARY KEY(Category_no)
);

CREATE TABLE BoardStory
(Board_no INT NOT NULL AUTO_INCREMENT COMMENT '게시판 번호',
Board_category_no INT COMMENT '게시판 카테고리',
Board_name CHAR(20) COMMENT '게시판 이름',
Board_content MEDIUMTEXT COMMENT '게시판 내용',
Board_writer_nickname CHAR(10) COMMENT '게시판 작성자',
Board_date TIMESTAMP COMMENT '게시판 작성 날짜',
Board_recommend_count INT DEFAULT 0 COMMENT '게시판 추천 수', 
PRIMARY KEY(Board_no, Board_category_no ),
FOREIGN KEY (Board_category_no) REFERENCES Category(Category_no) ON DELETE SET NULL ON UPDATE CASCADE
);
