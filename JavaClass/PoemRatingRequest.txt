package com.example.yja38.book;

import com.android.volley.Response;
import com.android.volley.toolbox.StringRequest;

import java.util.HashMap;
import java.util.Map;

/**
 * Created by SH on 2017-11-07.
 */

public class PoemRatingRequest extends StringRequest {

    final static private String URL="http://yja3806.dothome.co.kr/PoemRating.php";
    private Map<String,String> parameters;
    public PoemRatingRequest(String Member_nickname, String Poem_no,String Poem_grade, Response.Listener<String>listener){
        super(Method.POST,URL,listener,null);
        parameters=new HashMap<>();
        parameters.put("Member_nickname",Member_nickname);
        parameters.put("Poem_no",Poem_no);
        parameters.put("Poem_grade",Poem_grade);


    }

    @Override
    public Map<String,String> getParams(){
        return parameters;
    }
}
