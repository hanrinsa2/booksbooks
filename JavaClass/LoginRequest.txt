package com.example.yja38.book;

import com.android.volley.Response;
import com.android.volley.toolbox.StringRequest;

import java.util.HashMap;
import java.util.Map;

/**
 * Created by yja38 on 2017-10-27.
 */

public class LoginRequest extends StringRequest {

    final static private String URL="http://yja3806.dothome.co.kr/UserLogin.php";
    private Map<String,String> parameters;
    public LoginRequest(String Member_nickname, String Member_password, Response.Listener<String>listener){
        super(Method.POST,URL,listener,null);
        parameters=new HashMap<>();
        parameters.put("Member_nickname",Member_nickname);
        parameters.put("Member_password",Member_password);


    }

    @Override
    public Map<String,String> getParams(){
        return parameters;
    }




}
