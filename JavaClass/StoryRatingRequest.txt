package com.example.yja38.book;

import com.android.volley.Response;
import com.android.volley.toolbox.StringRequest;

import java.util.HashMap;
import java.util.Map;

/**
 * Created by SH on 2017-11-03.
 */

public class StoryRatingRequest extends StringRequest {
    final static private String URL="http://yja3806.dothome.co.kr/StoryRating.php";
    private Map<String,String> parameters;
    public StoryRatingRequest(String Member_nickname, String Story_no,String Story_grade, Response.Listener<String>listener){
        super(Method.POST,URL,listener,null);
        parameters=new HashMap<>();
        parameters.put("Member_nickname",Member_nickname);
        parameters.put("Story_no",Story_no);
        parameters.put("Story_grade",Story_grade);


    }

    @Override
    public Map<String,String> getParams(){
        return parameters;
    }



}
